    ```` javascript
    // 这里的逻辑关键在于用户有没有打开过红包设置面板点击选择过红包模版？(Y/N)
    //                   ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
    // Y: 必然插入新的红包模版到当前页面并且标记还原
    // N: 判断当前场景页面是否有后端返回的红包活动 id 数据？ (Y/N)
    //    ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
    // Y: 不插入红包单页模版
    //    - 因为保存过红包活动数据就会触发保存场景的数据
    //    - 所以当前场景必然会有红包模版数据和红包按钮
    // N: 判断是否有红包按钮存在？(Y/N)
    //    ^^^^^^^^^^^^^^^^^^^^^^^
    // Y: 不插入红包单页模版
    //    - 有可能当前场景是复制过来的
    // N: 插入红包单页模版
    if (lastSelectTplId) {
      lastSelectTplId = false;  <= 还原选择的红包模版 id 标记 (3 - 18)
      insert();
    } else {
      if (!curSceneTplMark && !hasRpButton) {
        insert();
      }
    }
    ````
